DFS(G)
	for each vertex u \in G.V
		u.Color = white
		u.Pi = NIL
	S = ø
	for each vertex u \in G.V
		if u.Color = white
			u.Color = grey
			push (S,u)
		while S != ø do
			u = pop (S)
			for each v \in adj(u)
				if v.Color = white then
					v.Color = grey
					v.Pi = u
					push(S,v)
			u.Color = black
